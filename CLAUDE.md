actorFilterEngine{
staticletshared=FilterEngine()privateletcontext:CIContext
privateletcolorSpace=CGColorSpace(name:CGColorSpace.linearSRGB)!privateinit(){
context=CIContext(options:[
.workingColorSpace:colorSpace,
.outputColorSpace:CGColorSpace(name:CGColorSpace.sRGB)!,
.useSoftwareRenderer:false,
.cacheIntermediates:true
])
}funcapply(_preset:FilterPreset,toimage:CIImage)->CIImage{
apply(preset.parameters,to:image)
}funcapply(_params:FilterParameters,toimage:CIImage)->CIImage{
varresult=image//Ordermattersforcoloraccuracy//1.Exposure&Contrast(beforecolor)
result=applyExposure(params.exposure,to:result)
result=applyContrast(params.contrast,to:result)//2.ToneCurve
ifparams.toneCurve!=.identity{
result=applyToneCurve(params.toneCurve,to:result)
}//3.Highlights&Shadows
result=applyHighlightsShadows(
highlights:params.highlights,
shadows:params.shadows,
to:result
)//4.WhiteBalance
result=applyWhiteBalance(
temperature:params.temperature,
tint:params.tint,
to:result
)//5.HSLAdjustments
ifparams.hsl!=.identity{
result=applyHSL(params.hsl,to:result)
}//6.Saturation&Vibrance
result=applySaturation(params.saturation,to:result)
result=applyVibrance(params.vibrance,to:result)//7.SplitTone
ifparams.splitTone!=.identity{
